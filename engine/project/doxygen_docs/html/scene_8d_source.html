45 if(name.length==0){
46 assert(0,"Failure, cannot query for a node name of length 0");
47 }
48 // Test if root node is our query, otherwise we'll
49 // proceed to perform DFS
50 if(name == GetRootNode.mNodeName){
51 return GetRootNode();
52 }
53
54 // Perform a depth-first traversal
55 ISceneNode current = mRootNode;
56 ISceneNode[] stack = current.mChildren;
57
58 while(stack.length > 0){
60 current = stack[$-1];
61 if(current.mNodeName == name){
62 return current;
63 }
64 // Remove the current node from stack
65 stack.popBack();
66 // Append all of the children to the back of the stack
67 stack ~= current.mChildren;
68 }
69
70 writeln("Node '"~name~"' could not be found! in 'FindNode'.");
71 return null;
72 }
