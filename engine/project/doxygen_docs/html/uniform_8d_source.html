65 void Transfer(){
66 if(mDataType=="int"){

68 }else if(mDataType=="float"){
69 glUniform1f(mCachedUniformLocation,cast(float)mPlainDataType);
70 }else if(mDataType=="vec2"){
71 vec2* v = cast(vec2*)mData;
72 glUniform2f(mCachedUniformLocation,v.data[0],v.data[1]);
73 }else if(mDataType=="vec3"){
74 vec3* v = cast(vec3*)mData;
75 glUniform3f(mCachedUniformLocation,v.data[0],v.data[1],v.data[2]);
76 }else if(mDataType=="mat4"){
77 mat4* m = cast(mat4*)mData;
78 glUniformMatrix4fv(mCachedUniformLocation, 1, GL_TRUE, m.DataPtr());
79 }else{
80 assert(0,"unsupported type, perhaps add more types in 'Transfer'?");
81 }
