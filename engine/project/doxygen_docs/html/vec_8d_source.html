325vec2 Normalize(vec2 v){

327 if(length == 0.0f){
328 writeln("WARNING: retrieved a zero value for magnitude");

375vec3 Cross(vec3 a, vec3 b){
376 vec3 result;
377
378 result[0] = a.y*b.z - b.y*a.z;
379 result[1] = a.z*b.x - b.z*a.x;
380 result[2] = a.x*b.y - b.x*a.y;
381
382 return result;
