71 void Input(){
72 // Store an SDL Event
73 SDL_Event event;
74 while(SDL_PollEvent(&event)){
75 if(event.type == SDL_QUIT){
76 writeln("Exit event triggered (probably clicked 'x' at top of the window)");
77 mGameIsRunning= false;
78 }
79 if(event.type == SDL_KEYDOWN){
80 if(event.key.keysym.scancode == SDL_SCANCODE_ESCAPE){
81 writeln("Pressed escape key and now exiting...");
82 mGameIsRunning= false;

84 else if(event.key.keysym.sym == SDLK_DOWN){
85 mCamera.MoveBackward();

88 mCamera.MoveForward();

91 mCamera.MoveLeft();
92 }
93 else if(event.key.keysym.sym == SDLK_RIGHT){
94 mCamera.MoveRight();

110 void SetupScene(){
111
112 // Create some nodes to attach to the SceneTree
113 // Geometry Data
114 GLfloat[] vertexData=
115 [
116 -0.5f, -0.5f, 0.0f, // Left vertex position
117 1.0f, 0.0f, 0.0f, // color
118 0.5f, -0.5f, 0.0f, // right vertex position
119 0.0f, 1.0f, 0.0f, // color
120 0.0f, 0.5f, 0.0f, // Top vertex position
121 0.0f, 0.0f, 1.0f, // color
122 ];
123
124 // Create a pipeline and associate it with a material
125 // that can be attached to meshes.
126 Pipeline basicPipeline = new Pipeline("basic","./pipelines/basic/basic.vert","./pipelines/basic/basic.frag");
127 IMaterial basicMaterial = new BasicMaterial("basic");
128
129 // Create an object and add it to our scene tree
130 ISurface triangle = new SurfaceTriangle(vertexData);
131 MeshNode m = new MeshNode("triangle",triangle,basicMaterial);
132 mSceneTree.GetRootNode().AddChildSceneNode(m);
133
134 // Add three uniforms to the basic material.
135 // The 4th parameter is set to the pointer where the value will be updated each frame.
136 // Becauses the model matrix will be different among models, then we will just leave
137 // this null for now.
138 basicMaterial.AddUniform(new Uniform("uModel", "mat4", null));
139 basicMaterial.AddUniform(new Uniform("uView", "mat4", mCamera.mViewMatrix.DataPtr()));
140 basicMaterial.AddUniform(new Uniform("uProjection", "mat4", mCamera.mProjectionMatrix.DataPtr()));
141
142 // Create a second object
143 GLfloat[] vertexData2 =
144 [
145 -0.5f, -0.5f, 0.0f, // Left vertex position
146 0.0f, 0.0f, // vertex texture(vt) coordinate
147 0.5f, -0.5f, 0.0f, // right vertex position
148 1.0f, 0.0f, // vertex texture(vt) coordinate
149 0.0f, 0.5f, 0.0f, // Top vertex position
150 0.5f, 1.0f, // vertex texture(vt) coordinate
151 ];
152 Pipeline texturePipeline = new Pipeline("texturePipeline","./pipelines/texture/basic.vert","./pipelines/texture/basic.frag");
153 IMaterial textureMaterial = new TextureMaterial("texturePipeline","./assets/sample.ppm");
154 textureMaterial.AddUniform(new Uniform("sampler1", 0));
155 textureMaterial.AddUniform(new Uniform("uModel", "mat4", null));
156 textureMaterial.AddUniform(new Uniform("uView", "mat4", mCamera.mViewMatrix.DataPtr()));
157 textureMaterial.AddUniform(new Uniform("uProjection", "mat4", mCamera.mProjectionMatrix.DataPtr()));
158
159 ISurface triangle2 = new SurfaceTexturedTriangle(vertexData2);
160 MeshNode m2 = new MeshNode("triangle2",triangle2,textureMaterial);
161 mSceneTree.GetRootNode().AddChildSceneNode(m2);
162 }

165 void Update(){
166 // A rotation value that 'updates' every frame to give some animation in our scene
167 static float yRotation = 0.0f; yRotation += 0.01f;
168
169 // Update our first object
170 MeshNode m = cast(MeshNode)mSceneTree.FindNode("triangle");
171 m.mModelMatrix = MatrixMakeTranslation(vec3(0.0f,0.0f,-1.0f));
172 m.mModelMatrix = m.mModelMatrix * MatrixMakeYRotation(yRotation);
173
174 // Update our second object
175 MeshNode m2 = cast(MeshNode)mSceneTree.FindNode("triangle2");
176 m2.mModelMatrix = MatrixMakeTranslation(vec3(0.0,-6.0f,-5.0f));
177 m2.mModelMatrix = m2.mModelMatrix * MatrixMakeScale(vec3(100.0f,100.0f,100.0f));
178 m2.mModelMatrix = m2.mModelMatrix * MatrixMakeXRotation(90.0.ToRadians);
179 }
