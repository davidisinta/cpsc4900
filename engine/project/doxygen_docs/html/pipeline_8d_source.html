26 GLuint CompilePipeline(string pipelineName, string vertexShaderSourceFilename, string fragmentShaderSourceFilename){
27 // Local nested function -- not meant for otherwise calling freely
28 void CheckShaderError(GLuint shaderObject){
29 // Retrieve the result of our compilation
30 int result;
31 // Our goal with glGetShaderiv is to retrieve the compilation status
32 glGetShaderiv(shaderObject, GL_COMPILE_STATUS, &result);
33
34 if(result == GL_FALSE){
35 int length;
36 glGetShaderiv(shaderObject, GL_INFO_LOG_LENGTH, &length);
37 GLchar[] errorMessages = new GLchar[length];
38 glGetShaderInfoLog(shaderObject, length, &length, errorMessages.ptr);
39 writeln(errorMessages);
40 }
41 }
42
43 // Compile our shaders
44 GLuint vertexShader;
45 GLuint fragmentShader;
46
47 // Use a string mixin to simply 'load' the text from a file into these
48 // strings that will otherwise be processed.
49 string vertexSource = readText(vertexShaderSourceFilename);
50 string fragmentSource = readText(fragmentShaderSourceFilename);
51
52 // Compile vertex shader
53 vertexShader = glCreateShader(GL_VERTEX_SHADER);
54 const char* vSource = vertexSource.ptr;
55 glShaderSource(vertexShader, 1, &vSource, null);
56 glCompileShader(vertexShader);
57 CheckShaderError(vertexShader);
58
59 // Compile fragment shader
60 fragmentShader= glCreateShader(GL_FRAGMENT_SHADER);
61 const char* fSource = fragmentSource.ptr;
62 glShaderSource(fragmentShader, 1, &fSource, null);
63 glCompileShader(fragmentShader);
64 CheckShaderError(fragmentShader);
65
66 // Create shader pipeline
67 mProgramObjectID = glCreateProgram();
68
69 // Link our two shader programs together.
70 // Consider this the equivalent of taking two .cpp files, and linking them into
71 // one executable file.
72 glAttachShader(mProgramObjectID,vertexShader);
73 glAttachShader(mProgramObjectID,fragmentShader);
74 glLinkProgram(mProgramObjectID);
75
76 // Validate our program
77 glValidateProgram(mProgramObjectID);
78
79 // Once our final program Object has been created, we can
80 // detach and then delete our individual shaders.
81 glDetachShader(mProgramObjectID,vertexShader);
82 glDetachShader(mProgramObjectID,fragmentShader);
83 // Delete the individual shaders once we are done
84 glDeleteShader(vertexShader);
85 glDeleteShader(fragmentShader);
86
87 // Store in the static pipeline map
88 mPipelineName = pipelineName;
89 sPipeline[mPipelineName] = mProgramObjectID;
90
91 // For debugging purposes, print out all information about the pipeline that has ben created
92 PrintShaderAttributesAndUniforms(mPipelineName,mProgramObjectID);
93
94 return mProgramObjectID;
95 }
96}

103static void PipelineUse(string name){
104 // First validate that the name is in the static map
105 GLint id = PipelineCheckValidName(name);
106
107 // Second, validate that the 'value' is indeed a graphics pipeline object.
108 if(glIsProgram(id) == GL_FALSE){
109 writeln("error: This shader '"~name~"' does not correspond to an active/valid pipeline");
110 writeln("This shader is: ",Pipeline.sPipeline[name]);
111 writeln("Candidates are: ", Pipeline.sPipeline.values());
112 assert(0,"Shader Use error");
113 }
114
115 // Activate our shader
116 glUseProgram(Pipeline.sPipeline[name]);
117}

211private string GL_type_to_string(GLenum type) {
212 switch(type) {
213 case GL_BOOL: return "bool";
214 case GL_INT: return "int";
215 case GL_FLOAT: return "float";
216 case GL_FLOAT_VEC2: return "vec2";
217 case GL_FLOAT_VEC3: return "vec3";
218 case GL_FLOAT_VEC4: return "vec4";
219 case GL_FLOAT_MAT2: return "mat2";
220 case GL_FLOAT_MAT3: return "mat3";
221 case GL_FLOAT_MAT4: return "mat4";
222 case GL_SAMPLER_2D: return "sampler2D";
223 case GL_SAMPLER_3D: return "sampler3D";
224 case GL_SAMPLER_CUBE: return "samplerCube";
225 case GL_SAMPLER_2D_SHADOW: return "sampler2DShadow";
226 default: break;
227 }
228 return "other";
229}
