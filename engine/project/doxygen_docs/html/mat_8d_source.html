557mat3 MatrixMakeRotation(float radians, vec3 vector){
558 vec3 v = vector.Normalize();
559 mat3 result = mat3(1.0f);
560
561 float cosa = cos(radians);
562 float sina = sin(radians);
563
564 vec3 temp = v * (1 - cosa);
565
566 result[0,0] = (cosa + temp.x * v.x);
567 result[0,1] = ( temp.x * v.y + sina * v.z);
568 result[0,2] = ( temp.x * v.z - sina * v.y);
569 result[1,0] = ( temp.y * v.x - sina * v.z);
570 result[1,1] = (cosa + temp.y * v.y);
571 result[1,2] = ( temp.y * v.z + sina * v.x);
572 result[2,0] = ( temp.z * v.x + sina * v.y);
573 result[2,1] = ( temp.z * v.y - sina * v.x);
574 result[2,2] = (cosa + temp.z * v.z);
575
576 return result;
577}

607mat4 MatrixMakePerspective(float fovy, float aspect_ratio, float zNear, float zFar) {
608 float f = 1.0f / tan(fovy / 2.0f);
609 float ar = aspect_ratio;
610 float nd = zNear;
611 float fd = zFar;
612
613 return mat4(
614 f / ar, 0, 0, 0,
615 0, f, 0, 0,
616 0, 0, (fd+nd)/(nd-fd), (2*fd*nd)/(nd-fd),
617 0, 0, -1, 0
618 );
619}
